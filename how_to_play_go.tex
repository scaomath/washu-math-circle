\newcommand\gostone[4][]{%
    \begin{scope}
        \fill[stone,#2-stone] (#3,#4) circle (0.45);
        \clip (#3,#4) circle (0.45);
        \shade[#2-highlight] (-0.15+#3,0.5+#4) circle (0.7);
    \end{scope}
    \node[#2-number] at (#3,#4) {\sffamily\bfseries{#1}};
}
